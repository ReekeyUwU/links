async function updateStatus() { try { const response = await fetch('https://api.allorigins.win/get?url=' + encodeURIComponent('https://api.lanyard.rest/v1/users/434389923869032458')); const resData = await response.json(); console.log("API Antwort:", resData); // <-- Das zeigt uns in der Konsole, was ankommt const data = JSON.parse(resData.contents); if (data.success) { const s = data.data.discord_status; const dot = document.getElementById('widget-dot'); const text = document.getElementById('widget-activity'); const statusColors = { 'online': '#43b581', 'idle': '#faa61a', 'dnd': '#f04747', 'offline': '#747f8d' }; if(dot) dot.style.backgroundColor = statusColors[s] || '#747f8d'; if(text) text.innerText = s.charAt(0).toUpperCase() + s.slice(1); } else { console.warn("API Erfolg war false:", data); } } catch(e) { console.error("Kritischer Fehler beim Abrufen:", e); } }
