async function updateStatus() { const userId = '434389923869032458'; try { const response = await fetch(`https://api.lanyard.rest/v1/users/${userId}`); const json = await response.json(); console.log("Lanyard Response:", json); // Schau hier nach! if (json.success) { const s = json.data.discord_status; const dot = document.getElementById('widget-dot'); const text = document.getElementById('widget-activity'); const colors = { 'online': '#43b581', 'idle': '#faa61a', 'dnd': '#f04747', 'offline': '#747f8d' }; if(dot) dot.style.backgroundColor = colors[s] || '#747f8d'; if(text) text.innerText = s.charAt(0).toUpperCase() + s.slice(1); } } catch(e) { console.error("Lanyard-Fehler:", e); } } updateStatus(); setInterval(updateStatus, 15000);
